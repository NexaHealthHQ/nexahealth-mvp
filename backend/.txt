import joblib

vectorizer = joblib.load("ml/vectorizer.joblib")
model = joblib.load("ml/risk_classifier.joblib")

def predict_risk(text: str) -> str:
    x = vectorizer.transform([text])
    prediction = model.predict(x)[0]
    return prediction
